## Example of Hook useFetch

```ts
import useFetch from "@src-path/hooks/useFetch";

...
async () => {
  const [data, isLoading, runFetch] = await useFetch({
    url: "http://example.com",
    method: "GET",
  }).then((data: any) => {
    return data;
  });
};
...

```
